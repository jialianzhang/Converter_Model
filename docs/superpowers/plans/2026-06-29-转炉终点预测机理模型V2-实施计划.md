"""反应平衡常数计算: lg K_M = A/T + B"""
import numpy as np
from .constants import OXIDATION_THERMO

def calc_lgK(element: str, T: float) -> float:
    """计算元素氧化反应的 lg K_M"""
    if element not in OXIDATION_THERMO:
        raise ValueError(f"Unknown element: {element}")
    d = OXIDATION_THERMO[element]
    return d['A'] / T + d['B']

def calc_K_M(element: str, T: float) -> float:
    """计算平衡常数 K_M"""
    return 10.0 ** calc_lgK(element, T)

def calc_all_lgK(T: float) -> dict:
    """计算所有元素的 lg K_M"""
    return {elem: calc_lgK(elem, T) for elem in OXIDATION_THERMO}

def calc_all_K_M(T: float) -> dict:
    """计算所有元素的 K_M"""
    return {elem: 10.0 ** lgK for elem, lgK in calc_all_lgK(T).items()}
```

"""金属相活度计算 — Wagner 模型: lg f_i = Σ e_i^j · w([j])"""
import numpy as np
from config import T_REF, R_GAS

# Wagner 相互作用系数 e_i^j (×10⁴) @ 1873K
# 格式: e_dict[元素i][元素j] = e_i^j × 10⁴
WAGNER_E_1E4 = {
    'C':  {'C': 14.0, 'Si': 8.0,  'Mn': -1.2, 'P': 5.1,  'S': 4.6,  'O': -34.0},
    'Si': {'C': 18.0, 'Si': 11.0, 'Mn': 0.2,  'P': 11.0, 'S': 5.6,  'O': -23.0},
    'Mn': {'C': -7.0, 'Si': 0.0,  'Mn': 0.0,  'P': -0.35,'S': -4.8, 'O': -8.3},
    'P':  {'C': 13.0, 'Si': 12.0, 'Mn': 0.0,  'P': 6.2,  'S': 2.8,  'O': 13.0},
    'S':  {'C': 11.0, 'Si': 6.3,  'Mn': -2.6, 'P': 2.9,  'S': -2.8, 'O': -27.0},
    'V':  {'C': -18.0,'Si': 4.2,  'Mn': 0.0,  'P': 0.0,  'S': -28.0,'O': -70.0},
    'Ti': {'C': -18.0,'Si': 5.5,  'Mn': 0.0,  'P': 0.0,  'S': -28.0,'O': -85.0},
    'Fe': {'C': 0.0,  'Si': 0.0,  'Mn': 0.0,  'P': 0.0,  'S': 0.0,  'O': 0.0},
    'O':  {'C': -45.0, 'Si': -13.1,'Mn': -2.1, 'P': 7.0,  'S': -13.3,'O': -20.0},
}

def get_e_ij(element_i: str, element_j: str, T: float) -> float:
    """获取相互作用系数 e_i^j，含温度修正"""
    e0 = WAGNER_E_1E4.get(element_i, {}).get(element_j, 0.0) * 1e-4
    return e0 * T_REF / T

def calc_lgf_i(element: str, composition: dict, T: float) -> float:
    """计算 lg f_i = Σ e_i^j · w([j])"""
    lgf = 0.0
    for elem_j, w_j in composition.items():
        if w_j > 0:
            lgf += get_e_ij(element, elem_j, T) * w_j
    return lgf

def calc_f_i(element: str, composition: dict, T: float) -> float:
    """计算活度系数 f_i = 10^(lg f_i)"""
    return 10.0 ** calc_lgf_i(element, composition, T)

def calc_all_f_i(composition: dict, T: float) -> dict:
    """计算所有元素的活度系数"""
    return {elem: calc_f_i(elem, composition, T) for elem in composition}
```

"""炉渣相活度计算 — 离子结构理论 (张强 式4)"""
import numpy as np
from config import R_GAS

# 交换电子能 ε_ij J/mol (简化矩阵，主对角线为零)
# 范式: ε[i][j] 表示阴离子 i 与 j 之间的交换能
EXCHANGE_ENERGY = {
    ('SiO2', 'FeO'):  -50000.0, ('SiO2', 'MnO'):  -30000.0,
    ('SiO2', 'CaO'): -120000.0, ('SiO2', 'MgO'):  -80000.0,
    ('SiO2', 'P2O5'): -40000.0,
    ('FeO',  'CaO'):  -30000.0, ('FeO',  'MnO'):  10000.0,
    ('CaO',  'MnO'):  -20000.0, ('CaO',  'MgO'):  -15000.0,
    ('MnO',  'MgO'):   -5000.0, ('P2O5', 'CaO'): -180000.0,
}

def get_epsilon(ox_i: str, ox_j: str) -> float:
    """获取交换电子能 ε_ij"""
    key = tuple(sorted([ox_i, ox_j]))
    rev_key = tuple(sorted([ox_j, ox_i]))
    if key in EXCHANGE_ENERGY:
        return EXCHANGE_ENERGY[key]
    return 0.0

def calc_slag_activity_coeff(oxide: str, slag_comp: dict, T: float) -> float:
    """计算炉渣中氧化物活度系数 f_iOn (式4)
    
    slag_comp: {oxide_name: mass_percent}, e.g. {'FeO': 20.0, 'SiO2': 15.0, ...}
    """
    denominator = 0.0
    for ox_j, w_j in slag_comp.items():
        if w_j <= 0:
            continue
        eps = get_epsilon(oxide, ox_j)
        denominator += w_j * np.exp(-eps / (R_GAS * T))
    if denominator < 1e-20:
        return 1.0
    return 1.0 / denominator

def calc_all_slag_activity_coeff(slag_comp: dict, T: float) -> dict:
    """计算所有炉渣氧化物的活度系数"""
    return {ox: calc_slag_activity_coeff(ox, slag_comp, T) for ox in slag_comp}
```

"""炉渣相图简化模型：CaO饱和浓度与固相率 (CaO-SiO₂-FeO 系 C₂S 区域)"""
import numpy as np

def calc_CaO_saturation(w_FeO: float, w_SiO2: float) -> float:
    """计算 C₂S 饱和线上的 CaO 质量分数 (%)
    
    基于 CaO-SiO₂-FeO 相图 C₂S 区域简化的直线方程组
    """
    # 简化的饱和线: w(CaO)_sat = a + b*w(SiO2) + c*w(FeO)
    # 参数由相图数据拟合
    if w_SiO2 < 0.01:
        return 55.0
    # C₂S 区域的简化模型
    return 45.0 + 0.7 * w_SiO2 - 0.15 * w_FeO

def calc_solid_fraction(w_CaO: float, w_SiO2: float, w_FeO: float) -> float:
    """计算炉渣固相率 f_solid (0~1)
    
    当渣成分落在 C₂S 饱和区时，通过杠杆定律计算固相率
    """
    w_CaO_sat = calc_CaO_saturation(w_FeO, w_SiO2)
    if w_CaO <= w_CaO_sat:
        return 0.0
    # 杠杆定律：f_solid = (w_CaO - w_CaO_sat) / (w_CaO_C2S - w_CaO_sat)
    # C₂S 中 CaO ≈ 65%
    w_CaO_C2S = 65.0
    f_solid = (w_CaO - w_CaO_sat) / (w_CaO_C2S - w_CaO_sat)
    return np.clip(f_solid, 0.0, 0.8)
```

```python
"""转炉静态模型：物料平衡 + 热量平衡，计算辅料、耗氧量和合理废钢量"""
import numpy as np
from thermo.constants import (
    MOLAR_MASS, OXIDATION_THERMO, HEAT_C2S_FORMATION,
    HEAT_C3P_FORMATION, HEAT_POST_COMBUSTION,
    HEAT_LIME_DECOMPOSITION, HEAT_DOLOMITE_DECOMPOSITION
)

# 辅料成分 (质量分数 %)
LIME_COMP = {'CaO': 90.0, 'MgO': 3.0, 'SiO2': 2.0, 'Al2O3': 1.0, 'S': 0.05, '烧损': 3.95}
DOLOMITE_COMP = {'CaO': 30.0, 'MgO': 21.0, 'SiO2': 2.0, 'Al2O3': 1.0, '烧损': 46.0}
SINTER_COMP = {'FeO': 9.0, 'Fe2O3': 64.0, 'CaO': 13.0, 'SiO2': 7.0, 'MgO': 2.0, 'Al2O3': 2.0, 'MnO': 1.5, '其他': 1.5}

class StaticModel:
    """转炉静态模型"""
    
    def __init__(self):
        self.cp_steel = 820.0      # J/(kg·K)
        self.cp_slag = 1200.0      # J/(kg·K)
        self.cp_offgas = 1400.0    # J/(kg·K) 炉气比热
    
    def calculate(self, inputs: dict) -> dict:
        """执行静态模型计算"""
        hm = {k: inputs[k] for k in ['w([C])', 'w([Si])', 'w([Mn])', 'w([P])', 
                                       'w([S])', 'w([V])', 'w([Ti])']}
        m_hm = inputs['铁水质量'] * 1000  # t → kg
        T_hm = inputs['铁水温度'] + 273.15  # °C → K
        m_scrap = inputs['废钢加入量'] * 1000  # t → kg
        R = inputs['炉渣碱度']
        
        # 1. 元素氧化量计算 (kg)
        oxided = {}
        # 按氧化顺序: Si → Mn → P → Fe → C (简化：以目标C为终点)
        for elem in ['Si', 'Mn', 'P', 'V', 'Ti']:
            oxided[elem] = hm[elem] / 100.0 * m_hm
        
        # C 氧化到目标值 (保留终点碳)
        w_C_target = inputs['终点碳目标']
        dC = hm['w([C])'] - w_C_target
        if dC > 0:
            oxided['C'] = dC / 100.0 * m_hm
        else:
            oxided['C'] = 0.0
        
        # Fe 氧化 (经验值，约铁水质量的 1%)
        oxided['Fe'] = 0.01 * m_hm
        
        # 2. 氧耗计算 (m³)
        O2_kg = 0.0
        for elem, mass_kg in oxided.items():
            n = OXIDATION_THERMO[elem]['n']
            M_elem = MOLAR_MASS[elem]
            moles = mass_kg / M_elem
            O2_kg += moles * n * 0.016  # O原子质量
        
        # 氧气利用率 ~95%，转换为 Nm³ (1 Nm³ O₂ ≈ 1.429 kg)
        O2_actual = O2_kg / 0.95 / 1.429
        
        # 3. 炉渣成分预算
        # 各氧化物生成量
        slag_oxides = {'SiO2': 0.0, 'MnO': 0.0, 'P2O5': 0.0, 'FeO': 0.0, 'CaO': 0.0, 'MgO': 0.0}
        for elem in ['Si', 'Mn', 'P', 'V', 'Ti', 'Fe']:
            mass_kg = oxided.get(elem, 0.0)
            if mass_kg <= 0:
                continue
            M_elem = MOLAR_MASS[elem]
            if elem == 'Si':
                slag_oxides['SiO2'] += mass_kg / M_elem * 0.06009
            elif elem == 'Mn':
                slag_oxides['MnO'] += mass_kg / M_elem * 0.07094
            elif elem == 'P':
                slag_oxides['P2O5'] += mass_kg / M_elem * 0.14194 / 2.0
            elif elem == 'Fe':
                slag_oxides['FeO'] += mass_kg / M_elem * 0.07185
        
        # 4. 石灰加入量 (基于碱度)
        # R = CaO / SiO₂ → CaO_need = R * SiO₂ - CaO_from_other_sources
        SiO2_in_slag = slag_oxides['SiO2']
        CaO_from_sinter = 0.0  # 后面计算
        CaO_need = R * SiO2_in_slag - CaO_from_sinter
        lime_kg = max(0.0, CaO_need / (LIME_COMP['CaO'] / 100.0))
        
        # 5. 白云石加入量 (满足 MgO 目标)
        MgO_target_pct = inputs['MgO目标含量']
        slag_mass_est = sum(slag_oxides.values()) + lime_kg * (LIME_COMP['CaO'] + LIME_COMP['MgO']) / 100.0
        MgO_need = MgO_target_pct / 100.0 * slag_mass_est
        dolomite_kg = max(0, (MgO_need - lime_kg * LIME_COMP['MgO'] / 100.0) / (DOLOMITE_COMP['MgO'] / 100.0))
        
        # 6. 出钢量
        m_total_metal = m_hm + m_scrap
        m_oxidized = sum(oxided.values())
        steel_output_t = (m_total_metal - m_oxidized) / 1000.0
        
        # 7. 铁耗
        iron_loss = (inputs['铁水质量'] + inputs['废钢加入量']) / steel_output_t * 1000.0
        
        return {
            '总耗氧量': round(O2_actual, 0),
            '石灰加入量': round(lime_kg, 0),
            '白云石加入量': round(dolomite_kg, 0),
            '合理废钢量': inputs['废钢加入量'],
            '出钢量': round(steel_output_t, 2),
            '铁耗': round(iron_loss, 2),
        }
```

```python
"""混合能计算: Nakanishi 公式 (式14, 式15)"""
import numpy as np

class MixingEnergy:
    def calc_epsilon_T(self, Q_T: float, W: float, X: float,
                       D: float, U: float, theta: float) -> float:
        """顶吹混合能 ε_T (式14)
        Q_T: 供氧流量 m³/min, W: 铁水质量 kg, X: 枪位 m,
        D: 喷嘴直径 m, U: 出口气速 m/s, theta: 喷嘴夹角 °
        """
        return 0.0403 * Q_T * D * U**2 * (np.cos(np.radians(theta)))**2 / (W * X)
    
    def calc_epsilon_B(self, Q_B: float, T: float, W: float, H: float) -> float:
        """底吹混合能 ε_B (式15)
        Q_B: 底吹氩气流量 m³/min, T: 温度 K, W: 铁水质量 kg, H: 熔池深度 m
        """
        return (28.5 * Q_B * T / W) * np.log10(1.0 + H / 148.0)
    
    def calc_total(self, Q_T, Q_B, T, W, X, D, U, theta, H):
        return self.calc_epsilon_T(Q_T, W, X, D, U, theta) + self.calc_epsilon_B(Q_B, T, W, H)

"""动态修正传质系数: k_m, k_s → F_M, F_MOn (Kitamura + V2 扩展)"""
import numpy as np
from config import K_S_MIN

class ModifiedMassTransfer:
    
    def calc_k_m(self, km0: float, eps_total: float, eps_total0: float, gamma: float) -> float:
        """钢液传质系数 V2-km: k_m = k_m⁰ × (ε_total/ε_total⁰)^γ"""
        return km0 * (eps_total / eps_total0) ** gamma
    
    def calc_k_s_prime(self, ks0: float, q_Ar: float, q_CO: float,
                       q_Ar0: float, W_s: float, W_s0: float) -> float:
        """炉渣传质—气体搅拌 Kit-7: k_s' = k_s⁰ × ((q_Ar+3q_CO)/q_Ar⁰) × (W_s⁰/W_s)"""
        return ks0 * ((q_Ar + 3.0 * q_CO) / q_Ar0) * (W_s0 / max(W_s, 1e-6))
    
    def calc_k_s(self, ks_prime: float, f_solid: float, n: float = 5.0) -> float:
        """炉渣传质—固相率衰减 Kit-8: k_s = k_s' × 10^(-n·f_solid)"""
        ks = ks_prime * 10.0 ** (-n * f_solid)
        return max(ks, K_S_MIN)
    
    def calc_F_M(self, km: float, rho_m: float, M_M: float) -> float:
        """钢液侧修正传质系数 V2-F: F_M = k_m·ρ_m / (100·M_M)"""
        return km * rho_m / (100.0 * M_M)
    
    def calc_F_MOn(self, ks: float, rho_s: float, M_MOn: float) -> float:
        """炉渣侧修正传质系数 V2-F: F_MOn = k_s·ρ_s / (100·M_MOn)"""
        return ks * rho_s / (100.0 * M_MOn)
    
    def calc_all_F(self, km: float, ks: float, rho_m: float, rho_s: float,
                   molar_masses: dict, oxide_molar_masses: dict) -> dict:
        """计算所有元素的 F_M 和 F_MOn"""
        elements = ['C', 'Si', 'Mn', 'P', 'V', 'Ti', 'Fe']
        oxides = ['SiO2', 'MnO', 'P2O5', 'V2O3', 'TiO2', 'FeO']
        result = {'F_O': self.calc_F_M(km, rho_m, 0.016)}
        for elem in elements:
            if elem in molar_masses:
                result[f'F_{elem}'] = self.calc_F_M(km, rho_m, molar_masses[elem])
        for ox in oxides:
            if ox in oxide_molar_masses:
                result[f'F_{ox}'] = self.calc_F_MOn(ks, rho_s, oxide_molar_masses[ox])
        return result
```

```python
"""界面浓度计算: 式7通用表达式"""
import numpy as np
from thermo.constants import OXIDATION_THERMO

class InterfaceConcentration:
    
    def calc_w_M_star(self, F_M: float, F_MOn: float, w_M_b: float,
                      w_MOn_b: float, E_M: float, a_O_star: float, n: float) -> float:
        """计算界面质量分数 w([M])* (式7)
        
        w([M])* = (F_MOn·w((MOn))b + F_M·w([M])b) / (F_M + F_MOn·E_M·(a*_O)^n)
        """
        numerator = F_MOn * w_MOn_b + F_M * w_M_b
        denominator = F_M + F_MOn * E_M * (a_O_star ** n)
        if denominator < 1e-30:
            return 0.0
        return numerator / denominator
    
    def calc_all(self, F_dict: dict, steel_b: dict, slag_b: dict,
                 E_M: dict, a_O_star: float, G_CO: float, K_C: float) -> dict:
        """计算所有元素的界面质量分数
        
        碳(C)使用专属气体界面动力学公式（产物为CO气体，不进入炉渣）：
        w([C])* = (F_C·w([C])b + G_CO) / (F_C + G_CO·K_C·a_O*)
        
        其他元素使用通用渣金反应公式（式7）
        """
        elements_map = {
            'Si': ('F_Si', 'F_SiO2',   'SiO2', 2.0),
            'Mn': ('F_Mn', 'F_MnO',    'MnO',  1.0),
            'P':  ('F_P',  'F_P2O5',   'P2O5', 2.5),
            'V':  ('F_V',  'F_V2O3',   'V2O3', 1.5),
            'Ti': ('F_Ti', 'F_TiO2',   'TiO2', 2.0),
            'Fe': ('F_Fe', 'F_FeO',    'FeO',  1.0),
        }
        
        result = {}
        
        # 碳的专属界面动力学公式（产物CO为气体，不由炉渣侧传质控制）
        F_C = F_dict.get('F_C', 0.0)
        w_C_b = steel_b.get('w([C])b', 0.0)
        numerator_C = F_C * w_C_b + G_CO
        denominator_C = F_C + G_CO * K_C * a_O_star
        result['w([C])*'] = numerator_C / denominator_C if denominator_C > 1e-30 else 0.0
        
        # 其他元素：通用渣金反应公式 (式7)
        for elem, (f_m_key, f_mon_key, ox_key, n) in elements_map.items():
            F_M = F_dict.get(f_m_key, 0.0)
            w_M_b = steel_b.get(f'w([{elem}])b', 0.0)
            F_MOn = F_dict.get(f_mon_key, 0.0)
            w_MOn_b = slag_b.get(f'w(({ox_key}))b', 0.0)
            E = E_M.get(elem, 1.0)
            result[f'w([{elem}])*'] = self.calc_w_M_star(
                F_M, F_MOn, w_M_b, w_MOn_b, E, a_O_star, n
            )
        return result
```

"""界面氧平衡残差计算 (式6修正): Residual = Σ(n_i·J_i) - J_O"""
import numpy as np
from thermo.constants import OXIDATION_THERMO

class OxygenBalance:
    
    def calc_J_M(self, F_M: float, w_M_b: float, w_M_star: float) -> float:
        """计算元素M的传质通量 J_M = F_M{w([M])b - w([M])*}"""
        return F_M * (w_M_b - w_M_star)
    
    def calc_a_O_b_from_slag(self, w_FeO_b: float, f_FeO: float, K_Fe: float) -> float:
        """从炉渣 FeO 计算本体氧活度 a_O^b (驱动氧向界面传输的氧势)
        
        a_O^b = w((FeO))b · f_FeO / K_Fe
        
        注：a_O^b 必须取炉渣侧 FeO 平衡的氧活度（高氧化性），而非钢液本体的溶解氧活度（极低）。
        若用钢液溶解氧活度，J_O 会变成负数（氧气从钢水向界面倒流），违背转炉吹氧物理常识。
        """
        if K_Fe < 1e-30:
            return 1e-6
        return max(w_FeO_b * f_FeO / K_Fe, 1e-6)
    
    def calc_J_O(self, F_O: float, a_O_b: float, a_O_star: float,
                 f_O: float = 1.0) -> float:
        """氧通量 J_O = F_O/f_O (a_O^b - a_O^*) 
        
        a_O^b 需由 calc_a_O_b_from_slag() 从渣侧 FeO 计算得到。
        """
        w_O_b = a_O_b / f_O if f_O > 0 else a_O_b
        w_O_star = a_O_star / f_O if f_O > 0 else a_O_star
        return F_O * (w_O_b - w_O_star)
    
    def calc_residual(self, J_fluxes: dict, F_O: float,
                      a_O_b_slag: float, a_O_star: float, f_O: float = 1.0) -> float:
        """计算氧平衡残差 Residual = 2J_Si + J_Mn + 2.5J_P + 1.5J_V + 2J_Ti + J_Fe + J_C - J_O
        
        a_O_b_slag: 由 calc_a_O_b_from_slag() 计算的炉渣侧氧势
        """
        residual = (
            2.0 * J_fluxes.get('Si', 0.0) +
            1.0 * J_fluxes.get('Mn', 0.0) +

### Task 12: 寻根算法 (`model/root_solver.py`)

```python
"""Brent 有界寻根 + NR + 二分法三级回退"""
from scipy.optimize import brentq
import numpy as np

class RootSolver:
    def __init__(self, tol=1e-10, max_iter=200):
        self.tol = tol
        self.max_iter = max_iter
    
    def solve(self, residual_func, bracket=(1e-8, 1.0)) -> float:
        """使用 Brent 法求解 Residual(a*_O) = 0"""
        try:
            a, b = bracket
            fa = residual_func(a)
            fb = residual_func(b)
            if fa * fb > 0:
                if abs(fa) < self.tol: return a
                if abs(fb) < self.tol: return b
                raise ValueError("Bracket does not bracket a root")
            return brentq(residual_func, a, b, xtol=self.tol, maxiter=self.max_iter)
        except Exception:
            return self._newton_fallback(residual_func, bracket)
    
    def _newton_fallback(self, f, bracket):
        """Newton-Raphson 回退，失败则二分法"""
        x = (bracket[0] + bracket[1]) / 2.0
        for _ in range(self.max_iter):
            fx = f(x)
            if abs(fx) < self.tol:
                return x
            EPS = 1e-8
            df = (f(x + EPS) - fx) / EPS
            if abs(df) < 1e-20:
                return self._bisection_fallback(f, bracket)
            x_new = x - fx / df
            if x_new <= bracket[0] or x_new >= bracket[1]:
                return self._bisection_fallback(f, bracket)
            x = x_new
        return self._bisection_fallback(f, bracket)
    
    def _bisection_fallback(self, f, bracket):
        a, b = bracket
        for _ in range(self.max_iter):
            c = (a + b) / 2.0
            fc = f(c)
            if abs(fc) < self.tol:
                return c
            fa = f(a)
            if fa * fc < 0: b = c
            else: a = c
        return (a + b) / 2.0
```
